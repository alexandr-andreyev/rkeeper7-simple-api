Для сохранения заказа можете попробовать выполнять запрос SaveOrder и указывать необходимые для задачи атрибуты: remindTime="" readyTime="" printTime=""

<?xml version="1.0" encoding="UTF-8"?>
<RK7Query>
    <RK7CMD CMD="SaveOrder">
        <Order guid="{499247FA-9AB6-4AB0-B525-B747DCC33251}"/>
        <Session remindTime="" readyTime="" printTime="" >
            <Dish code="124" quantity="2000"/>
        </Session>
    </RK7CMD>
</RK7Query>
Где:
remindTime="" Время начала готовки блюд пакета. Если не задано, то используется текущее время

readyTime="" Время, к которому должны быть приготовлены все блюда пакета

printTime="" Время первой сервис-печати.

И/или использовать комментарии к заказу:

<?xml version="1.0" encoding="UTF-8"?>
<RK7Query >
    <RK7CMD CMD="CreateOrder" >
        <Order persistentComment="" nonPersistentComment="">
            <Table id=""/>
            <OrderType id=""/>
        </Order>        
    </RK7CMD>
</RK7Query>
persistentComment="" - Сохраняемый комментарий
nonPersistentComment="" - Несохраняемый комментарий

Можете в запросе еще указывать тип заказа:
<OrderType id=""/>
